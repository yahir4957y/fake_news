# Herramientas necesarias en el sistema:
# - Python 3.10 o superior
# - pip

# Instalacion sugerida dentro del entorno del proyecto backend:
# Ejecutar desde la raiz del proyecto:
# cd backend
# python -m venv .venv
# .\.venv\Scripts\Activate.ps1
# pip install fastapi "uvicorn[standard]" python-multipart pillow python-dotenv google-generativeai psycopg2-binary "python-jose[cryptography]" requests pydantic
